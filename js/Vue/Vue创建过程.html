  2. 总结

    • Vue.prototype 下的属性和方法的挂载主要是在 src/core/instance 目录中的代码处理的

    我们把关注点放在 data的响应式是如何实现的，我们看initdata()函数内部。主要做了两件事

    • "观察" 顶层数据(根对象，即构造时我们传入的对象)

      observe(data, true /* asRootData */)

      我们点进这个函数。

      export function observe (value: any, asRootData: ?boolean): Observer | void {
  if (!isObject(value) || value instanceof VNode) {
    return
  }
  let ob: Observer | void
  // 如果已经添加过 observer
  if (hasOwn(value, '__ob__') && value.__ob__ instanceof Observer) {
    ob = value.__ob__
    // 没有就new一个
  } else if (
    shouldObserve &&
    !isServerRendering() &&
    (Array.isArray(value) || isPlainObject(value)) &&
    Object.isExtensible(value) &&
    !value._isVue
  ) {
    ob = new Observer(value)
  }
  if (asRootData && ob) {
    ob.vmCount++
  }
  return ob
}

      按照流程我们应该执行else分支。即

      ob = new Observer(value)  // value 是根对象

      我们看看 Observer是如何构造的

      constructor (value: any) {
  this.value = value
  this.dep = new Dep()
  this.vmCount = 0
  // 将 observer实例绑定到根对象的__ob__属性上。
  def(value, '__ob__', this)
  // 如果是数组
  if (Array.isArray(value)) {
    if (hasProto) {
      protoAugment(value, arrayMethods)
    } else {
      copyAugment(value, arrayMethods, arrayKeys)
    }
    this.observeArray(value)
  } else {
  // 如果是对象
    this.walk(value)
  }
}
// 对对象的每个属性进行 defineReactive
walk (obj: Object) {
  const keys = Object.keys(obj)
  for (let i = 0; i < keys.length; i++) {
    defineReactive(obj, keys[i])
  }
}

      先不管数组，我们再来看看这个 defineReactive 。我们把关注点放在下面。

      Object.defineProperty(obj, key, {
  enumerable: true,
  configurable: true,
  get: function reactiveGetter () {
    const value = getter ? getter.call(obj) : val   // 获取新值
    // Dep.target 是Dep的静态成员，可以把它看成开关
    if (Dep.target) {
      dep.depend()
      // 子对象的observer
      if (childOb) {
        childOb.dep.depend()
        if (Array.isArray(value)) {
          dependArray(value)
        }
      }
    }
    return value
  },
  set: function reactiveSetter (newVal) {
    const value = getter ? getter.call(obj) : val   // 获取旧值
    childOb = !shallow && observe(newVal)           // 观察新值
    dep.notify()      															// 通知
  }
})

      其中, dep.depend() 的作用是 添加依赖(调用Watcher.addDep())，并且过滤掉已经添加过的。现在我们看看 Watcher 。

      export default class Watcher {
  constructor (vm: Component, expOrFn: string | Function, cb: Function
  ) {
    this.vm = vm
		this.getter = parsePath(expOrFn)
    this.cb = cb
    this.value = this.get()
  }
  // 访问数据
  get () {
    pushTarget(this)
    let value
    const vm = this.vm
    value = this.getter.call(vm, vm)
    popTarget()
    // 清除所有依赖
    this.cleanupDeps()
    return value
  }
  //添加依赖
  addDep (dep: Dep) {
    const id = dep.id
    if (!this.newDepIds.has(id)) {
      this.newDepIds.add(id)
      this.newDeps.push(dep)
      if (!this.depIds.has(id)) {
        dep.addSub(this)
      }
    }
  }
  // 更新数据，根据回调函数渲染
  update () {
    const oldValue = this.value
    this.value = this.get()
    this.cb.call(this.vm, this.value, oldValue)
  }
