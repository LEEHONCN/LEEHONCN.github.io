# Tapable

基于 发布-订阅 模式。

从使用开始，我们是怎么使用 tapable 的呢？

const hook = new SyncHook();
hook.tap("A", ()=>console.log("调用call来显示我"));
console.log("我是第一句哦");
hook.tap({
        name: "B",
        before: "A"
    },() => console.log("我是第二句哦"));
// 调用call 注册过的回调函数才会执行
hook.call();

// 我是第一句哦
// 我是第二句哦
// 调用call来显示我

所有钩子类都继承Hook。先来看Hook。

withOptions(options) {
		const mergeOptions = opt =>
			Object.assign({}, options, typeof opt === "string" ? { name: opt } : opt);

		// Prevent creating endless prototype chains
		options = Object.assign({}, options, this._withOptions);
		const base = this._withOptionsBase || this;
		const newHook = Object.create(base);

		newHook.tap = (opt, fn) => base.tap(mergeOptions(opt), fn);
		newHook.tapAsync = (opt, fn) => base.tapAsync(mergeOptions(opt), fn);
		newHook.tapPromise = (opt, fn) => base.tapPromise(mergeOptions(opt), fn);
		newHook._withOptions = options;
		newHook._withOptionsBase = base;
		return newHook;
}

# call

其实 执行call 就是执行 compile(),也就是被重写的。只不过传入了几个参数

Object.defineProperties(Hook.prototype, {
	_call: {
		value: createCompileDelegate("call", "sync"),
		configurable: true,
		writable: true
	},
	_callAsync: {
		value: createCompileDelegate("callAsync", "async"),
		configurable: true,
		writable: true
	},
	_promise: {
		value: createCompileDelegate("promise", "promise"),
		configurable: true,
		writable: true
	}
});

function createCompileDelegate(name, type) {
	return function lazyCompileHook(...args) {
		this[name] = this._createCall(type);
		return this[name](...args);
	};
}

_createCall(type) {
  return this.compile({
    taps: this.taps,
    interceptors: this.interceptors,
    args: this._args,
    type: type
  });
}

# HookCodeFactory 类

每个 Hook根据不同的需求 可以重写 content 方法实现 用来生产需要的 call callasync promise 方法。

它使我们构造Hook时传入的参数(字符串数组)， 变成生成的函数的参数列表。这样我们可以在call中传入参数，在回调函数中使用参数。

const hook = new SyncHook(["C"]);
hook.tap("A", (C)=>console.log("调用call来显示我 " + C));
hook.tap({
        name: "B",
        before: "A"
    },(C) => {
    console.log("我是第二句哦 "+C)
});
console.log("我是第一句哦");
hook.call("传入");

// 查看生成的函数
console.log(hook.call.toString())
