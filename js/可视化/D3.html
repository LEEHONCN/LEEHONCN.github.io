// 选择器 返回 选择集
// 选择集：选择器返回的元素集合。
selection = d3.select();
selection = d3.selectAll();
// 查看状态
selection.empty();  // 如果为空返回true
selection.node();   // 返回第一个元素
selection.size();   // 选择集的大小
// 设置或获取属性
selection.attr(name,[value]);  // 设置属性 返回属性值
selection.classed(name,[value]);  // 设置类
selection.style(name,[value]);  // 设置样式
selection.property(name,[value]);  // 返回属性名 返回属性值	弥补 attr
selection.text(name,[value]);  // 设置文本 返回文本
selection.html(name,[value]);  // 设置或返回 html
// 添加，插入，删除
selection.append();
selection.insert();
selection.remove();
// 数据绑定
selection.datum();   // 统一绑定
selectio.data();     // 分别绑定
// 选择集处理数据绑定多余的元素和数据
selection.enter().append();
selection.exit().remove();

selection.filter(func);    // 根据条件选择选择集的子集
// 排序
selection.sort(func);   // 将选择集按照规则排序
// 遍历
selection.each(func);
// 借用
selection.call(func);    // 将自身作为参数传给 函数
